\section{Introduction}


\begin{exercise}{4}
Let $x_1, x_2, x_3, \dots$ be an infinite collection of independent variables indexed by the natural numbers.
A polynomial with coefficients in a field $k$ in the $x_i$ is a finite linear combination of (finite) monomials $x_{i_1}^{e_1}, \dots, x_{i_n}^{e_n}$.
Let $R$ denote the set of all polynomials in the $x_i$.
Note that we can add and multiply elements of $R$ in the usual way.
Thus, $R$ is the polynomial ring $k[x_1, x_2, \dots]$ in infinitely many variables.
\begin{enumerate}
    \item Let $I = \brackets{x_1, x_2, x_2, \dots}$ be the set of polynomials of the form $x_{t_1}f_1 + \dots + x_{t_m}f_m$, where $f_j \in R$.
    Show that $I$ is an ideal in the ring $R$.
    \item Show, arguing by contradiction, that $I$ has no finite generating set.
    Hint: it is not enough only to consider subsets of $\set{x_i \mid i \geq 1}$.
\end{enumerate}
\end{exercise}
\begin{proof}
\begin{enumerate}
    \item Choose a finite set of monomials $x_{t_1}, \dots, x_{t_m}$, we have that $0x_{t_1}+ \dots + 0x_{t_m} = 0$, so that $0 \in I$.
    Furthermore, let $g, g' \in I$, thus, $g = x_{t_1}f_1 + \dots + x_{t_m}f_m$
    and $g' = x_{t_1}'f_1' + \dots + x_{t_n}'f_n'$.
    We then have $g + g' = x_{t_1}f_1 + \dots + x_{t_m}f_m + x_{t_1}'f_1' + \dots + x_{t_n}'f_n'$.
    In some cases we are allowed to accumulate $f_i$ and $f_j$, if $x_{t_i} = x_{t_j}'$, however this does not change the fact that $g + g' \in I$.
    Lastly, consider any polynomial $h \in R$.
    We have $hg = h(x_{t_1}f_1 + \dots + x_{t_m}f_m) = x_{t_1}(f_1 h) + \dots + x_{t_m}(f_m h)$, which belongs to $I$, so that $I$ fulfills the absorption property, proving that $I$ is an ideal.
    \item Following the hint, suppose, for the sake of contradiction, that $I$ has a finite generating set: $f_1, \dots, f_n$, where $f_i$ are finite linear combinations of finite products of monomials.
    Now consider the ideal $I'$, generated by all the variables that compose each $f_i$.
    Since we have a finite number of $f_i$, each of which is composed of finite sums and products, the result list must be finite.
    Thus $I \subseteq I' = \brackets{x_1,\dots,x_n}$.
    Notice, however, that $x_{n+1} \notin I = \brackets{x_1,\dots, x_n, x_{n+1}, \dots}$, but $I \subseteq I'$, which produces a contradiction.
    Hence, it must be the case that $I$ has no finite set of generators.
\end{enumerate}
\end{proof}
