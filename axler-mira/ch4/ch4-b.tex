\section{4B Derivatives of integrals}
\addcontentsline{toc}{section}{4B Derivatives of integrals}


\begin{exercise}{3}
Suppose $f: \R \to \R$ is a Lebesgue measurable function such that $f^2 \in \cL(\R)$.
Prove that
\begin{align*}
    \lim_{t \downarrow 0} \frac{1}{2t} \int_{b-t}^{b+t} \absoluteValue{f - f(b)}^2 = 0
\end{align*}
for almost every $b \in \R$.
\end{exercise}
\begin{proof}
To prove this result, we need the following Lemma:
If $\int_{[a,b]} \absoluteValue{f}^2 d\lambda < \infty$, then $\int_{[a,b]} \absoluteValue{f} d\lambda < \infty$.

First, if $\absoluteValue{f(x)} \geq 1$, then the result is clear, as $\absoluteValue{f(x)} \leq \absoluteValue{f(x)}^2$, and the result follows by the order Theorem of Lebesgue integration (3.22).
Thus, we assume $\absoluteValue{f(x)} < 1$.
There exists a $c \in [0,1]$, such that $\absoluteValue{f(x)} \leq c + \absoluteValue{f(x)}^2$, we then have
\begin{align*}
    \int_{[a,b]} \absoluteValue{f}
    \leq \int_{[a,b]} c + \absoluteValue{f}
    = c(a-b) + \int_{[a,b]} \absoluteValue{f}
    < \infty.
\end{align*}
Completing the proof of the Lemma requires separating $f$ into two functions $g_1(x) = f(x)$ if $\absoluteValue{f(x)} \leq \absoluteValue{f(x)}^2$ and 0 otherwise and $g_2(x) = f(x)$ if $\absoluteValue{f(x)} > \absoluteValue{f(x)}^2$ and 0 otherwise.
The result would follow by the additivity of the integrals, and the fact that both integrals are finite, completing the proof of the Lemma.

For the main result, let $g = \absoluteValue{f-f(b)}^2 \leq \absoluteValue{f}^2 + \absoluteValue{2f(b)f} + \absoluteValue{f(b)}^2$, which by the previous result has a finite integral so that $g \in \cL^1(\R)$ by the order Theorem of Lebesgue integrals (3.22).
Knowing that $g \in \cL^1(\R)$ lets us apply the Lebesgue differentiation Theorem (4.10), completing the proof.
\end{proof} 

\begin{exercise}{6}
Prove that if $h \in \cL^1(\R)$ and $\int_\infty^s h = 0$ for all $s \in \R$, then $h(s) = 0$ for almost every $s \in \R$.
\end{exercise}
\begin{proof}
Let $g(x) = \int_{-\infty}^x h$.
By the Lebesgue differentiation Theorem (4.19), we have that $g'(b) = h(b)$ for a.e. $b$.
Since $g(b) = 0$ (at a.e. $b$), then $g'(b) = 0$ (at a.e. $b$), and thus $h(b) = 0$ (at a.e. $b$), completing the proof.
\end{proof} 

\begin{exercise}{7}
Give an example of a Borel subset of $\R$ whose density at 0 is not defined.
\end{exercise}
\begin{proof}
Consider the set given by 
\begin{align*}
    E = \bigcup \sqrBrackets{\frac{1}{3^{2n+1}}}, \frac{1}{3^{2n}}.
\end{align*}

The density of $E$ at 0 is defined by 
\begin{align*}
    \lim_{t \downarrow 0} \frac{\absoluteValue{E \cap (-t, t)}}{2t}.
\end{align*}

For $t = 3^{-2m}$, where $m \in \N$, we have 
\begin{align*}
    \absoluteValue{E \cap (-3^{-2m}, 3^{-2m})} 
    =& \sum_{k=m}^\infty 3^{-2k} - 3^{-(2k+1)}\\
    =& \sum_{k=m}^\infty 3^{-2k} (1 - 1/3)\\
    =& \frac{2}{3}\sum_{k=m}^\infty 9^{-k}\\
    =& \frac{2}{3}\frac{9^{-m}}{1-9^{-1}}\\
    =& \frac{2}{3} \cdot \frac{9}{8} \cdot 3^{-2m}\\
    =& \frac{3}{4} \cdot 3^{-2m},
\end{align*}
so that the density evaluates to $3/8$.

We will now do the same process for $t = 3^{-(2m+1)}$.
Notice that in this case, $t$ is the left end point of the interval, so that $3^{-(2m+1)} - 3^{-2m}$ is excluded from the measure.
This gives us a similar calculation as above, where we shift the summation index:
\begin{align*}
    \absoluteValue{E \cap (-3^{-(2m+1)}, 3^{-(2m+1)})} 
    =& \sum_{k=m+1}^\infty 3^{-2k} - 3^{-(2k+1)}\\
    =& \frac{2}{3}\sum_{k=m+1}^\infty 9^{-k}\\
    =& \frac{2}{3}\frac{9^{-(m+1)}}{1-9^{-1}}\\
    =& \frac{2}{3} \cdot \frac{9}{8} \cdot 3^{-2(m+1)}\\
    =& \frac{3}{4} \cdot 3^{-2(m+1)},
\end{align*}
and now the density evaluates to $(3/8)(3^{-2(m+1)}/3^{-(2m+1)}) = 1/8$.

This means that the density at 0 oscillates between $1/8$ and $3/8$, so that the limit does not exist, completing the proof.
\end{proof} 

\begin{exercise}{9}
Prove that if $c \in [0, 1]$, then there exists a Borel set $E \subseteq \R$ such that the density of $E$ at 0 is $c$.
\end{exercise}
\begin{proof}
This proof is due to Davide Ravotti.
Let $c \in [0,1]$.
For an interval $I_n = [1/(n+1), 1/n]$, let $c_n \in I_n$ denote the point such that $\absoluteValue{[1/(n+1), c_n]} = c\absoluteValue{I_n}$.
Now let 
\begin{align*}
    E = \bigcup \sqrBrackets{-c_n, -\frac{1}{n+1}} \cup \sqrBrackets{\frac{1}{n+1}, c_n}. 
\end{align*}
Notice that for any $n$, we have
\begin{align*}
    \absoluteValue{E \cap \parens{0, \frac{1}{n}}}
    = \sum_{k \geq n} \absoluteValue{E \cap I_k}
    = \sum_{k \geq n} c \absoluteValue{I_k}
    = \frac{c}{n},
\end{align*}
so that for any $n \in \N$, we have $\absoluteValue{E \cap (-1/n, 1/n)} = 2c/n$.

Fix $t \in (0,1)$ and let $n \in \N$ be such that $1/(n+1) < t \leq 1/n$.
We have 
\begin{align*}
    \frac{\absoluteValue{E \cap (-t, t)}}{2t}
    \leq \frac{\absoluteValue{E \cap (-1/n, 1/n)}}{2/(n+1)}
    = \frac{2c/n}{2/(n+1)}
    = c\frac{n+1}{n}.
\end{align*}
Similarly,
\begin{align*}
    \frac{\absoluteValue{E \cap (-t, t)}}{2t}
    \geq \frac{\absoluteValue{E \cap (-1/(n+1), 1/(n+1))}}{2/n}
    = \frac{2c/(n+1)}{2/n}
    = c\frac{n}{n+1}.
\end{align*}
By the sandwich Theorem of limits, we have that the limit as $t \to 0$ exists and
\begin{align*}
    \lim_{t \downarrow 0} \frac{\absoluteValue{E \cap (-t, t)}}{2t} 
    = c,
\end{align*}
completing the proof.
\end{proof}
