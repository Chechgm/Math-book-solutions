\begin{exercise}{1}
Give an example of a sequence $f_1, f_2, \dots$ of functions from $\N$ to $[0, \infty)$ such that $\lim_{k \to \infty} f_k(m) = 0$ for every $m \in \N$ but $\lim_{k \to \infty} \int f_k d\mu = 1$ where $\mu$ is counting measure on $\N$.
\end{exercise}
\begin{proof}
Consider the sequence given by $f_k(m) = 1$ if $m = k$ and 0 otherwise.
We see that for any $m$, $\lim_{k \to \infty} f_k(m) = 0$ but also $\int f_k d\mu = 1$ under the counting measure, so that $\lim f_k d\mu = 1$, completing the proof.
Notice that this even though $g(m) = 1$ dominates every $f_k$, the dominated convergence Theorem does not apply, as $g$ has no finite integral.
\end{proof} 

\begin{exercise}{2}
Give an example of a sequence $f_1, f_2, \dots$ of continuous functions from $\R$ to $[0,1]$ such that $\lim_{k \to \infty} f_k(x) = 0$ for every $x \in \R$ but $\lim_{k \to \infty} f_k d\lambda = \infty$, where $\lambda$ is Lebesgue measure on $\R$
\end{exercise}
\begin{proof}
Let 
\begin{align*}
    f_k(x) = 
    \begin{cases}
        0   &\text{ if }    x < k\\
        x-k &\text{ if }  x \in [k, k+1]\\
        1   &\text{ if }    x > k+1.
    \end{cases}
\end{align*}
Each function in this sequence is continuous.
Moreover, we have that for all $x$, $\lim_{k \to \infty} f_k(x) = 0$, but also for all $k$, $\int f_k d\lambda = \infty$, so that $\lim_{k \to \infty} f_k d\lambda = \infty$, completing the proof.
\end{proof} 

\begin{exercise}{3}
Suppose $\lambda$ is Lebesgue measure on $\R$ and $f: \R \to \R$ is a Borel measurable function such that $\int \absoluteValue{f} d\lambda < \infty$.
Define $g: \R \to \R$ by $g(x) = \int_(-\infty, x) f d\lambda$.
Prove that $g$ is uniformly continuous on $\R$.
\end{exercise}
\begin{proof}
We have 
\begin{align*}
    \absoluteValue{g(x) - g(y)}
    =& \absoluteValue{\int_{(-\infty, x)}f d\lambda - \int_{(-\infty, y)}f d\lambda}\\
    =& \absoluteValue{\int f\chi_{(-\infty, x)} d\lambda - \int f\chi_{(-\infty, y)} d\lambda}\\
    =& \absoluteValue{\int f\chi_{(-\infty, x)} - f\chi_{(-\infty, y)} d\lambda}\\
    =& \absoluteValue{\int f\chi_{[x, y)} d\lambda}\\
    =& \absoluteValue{\int_{[x, y)} f d\lambda}\\
    \leq& \int_{[x, y)}\absoluteValue{f} d\lambda,
\end{align*}
where the last inequality follows from 2.45, and we have assumed, without loss of generality, that $x < y$.

Let $\epsilon > 0$.
Since we assume $\int \absoluteValue{f} d\lambda < \infty$, we can use 3.29 to conclude there is a $\delta > 0$ such that $\int_B \absoluteValue{f} d\lambda < \epsilon$ for all Borel $B$ with $\lambda(B) < \delta$.
translating this to our result above, we have that for any $x,y \in \R$ with $\absoluteValue{x - y} < \delta$, we get that $\absoluteValue{g(x) - g(y)} \leq \int_{[x, y)} \absoluteValue{f} d\lambda < \epsilon$, giving us the desired uniform continuity.
\end{proof} 

\begin{exercise}{6}
Let $\lambda$ denote Lebesgue measure on $\R$.
Give an example of a continuous function $f: [0, \infty) \to \R$ such that $\lim_{t \to \infty} \int_{[0,t]} f d\lambda$ exists (in $\R$) but $\int_{[0, \infty)} f d\lambda$ is not defined.
\end{exercise}
\begin{proof}
An example of such function is $f(x) = \sin x/x$ and $f(0) = 0$.
We will prove this is the case in steps.
First, notice that integrating by parts (with $u = 1/x$ and $dv = \sin x$, so that $du = -dx/x^2$ and $v = -\cos x$), we get
\begin{align*}
    \int_a^b \frac{\sin x}{x} dx 
    = \frac{\cos a}{a} - \frac{\cos b}{b} - \int_a^b \frac{\cos x}{x^2} dx.
\end{align*}
Since $f(x)$ (on the left hand side of the equality) is continuous on $[0,1]$ then it is integrable.
On the other hand, taking the limit as $b \to \infty$, and looking instead at the absolute value of $\cos$ (which will not affect our results, since the integral will turn out to be finite) gives us 
\begin{align*}
    \frac{\absoluteValue{\cos a}}{a} - \int_a^\infty \frac{\absoluteValue{\cos x}}{x^2} dx
    \leq \frac{\absoluteValue{\cos a}}{a} - \int_a^\infty \frac{1}{x^2} dx,
\end{align*}
so that by the comparison test, the integral is finite in as $b \to \infty$.

Now by the linearity of integrals, we have
\begin{align*}
    \int_0^\infty \frac{\sin x}{x} dx
    = \sum_{n \in \N} \int_{n\pi}^{(n+1)\pi} \frac{\sin x}{x} dx.
\end{align*}
Letting $y = (x - n\pi)/\pi$ so that $dy = dx/\pi$, we obtain for the positive part of $\sin x$:
\begin{align*}
    \sum_{2n \in \N} \int_{n\pi}^{(n+1)\pi} \frac{\sin x}{x} dx
    =& \sum_{2n \in \N} \int_0^1 \frac{\sin (\pi y + \pi n)}{\pi y + \pi n} \pi dy\\
    =& \sum_{2n \in \N} \int_0^1 \frac{\sin (\pi y)}{y + n} dy\\
    =& \sum_{2n \in \N} \frac{\cos \pi y}{1 + n}\bigg|_0^1,
\end{align*}
the harmonic series, which we know diverges.
Likewise, for the negative part of $\sin x$ (summing over $2n+1 \in \N$, we would obtain $-\infty$, so that the integral of $f$ is not defined on $[0,\infty)$, completing the proof.
\end{proof} 

\begin{exercise}{8}
Let $f: [a,b] \to \R$ be a bounded function and $P_n$ be a partition that divides $[a,b]$ into $2^n$ subintervals of equal size with $I_1, I_2, \dots, I_{2^n}$ the corresponding closed subintervals.
Define $g_n = \sum_{j=1}^{2^n} (\inf_{I_j}) \chi_{I_j}$, $h_n = \sum_{j=1}^{2^n} (\sup_{I_j}) \chi_{I_j}$, and the limits of these functions as $f^L(x) = \lim g_n(x)$, and $f^U(x) = \lim h_n(x)$.
Prove that $\set{x \in [a,b]: f^U(x) \neq f^L(x)} = \set{x \in [a,b]: f \text{ is not continuous at } x}$.
\end{exercise}
\begin{proof}
We will prove this by double containment.

($\subseteq$)
We will prove the first part by contrapositive.
For that matter, suppose $f$ is continuous at $x \in [a,b]$.
Thus, for all sequences $(x_n)$ with $x_n \to x$, we have $f(x_n) \to f(x)$.
%, which also means that for all $\epsilon > 0$, there exists an $N \in \N$ so that whenever $n > N$, it holds that $f(x_n) \in B_\epsilon(f(x))$.
In particular, consider the two following sequences:
first $(x_n)$, where $x_n$ is chosen from the $I_j^n$ (we include the superscript here to make it clear that $I_j$ depends on $n$) in $P_n$ so that $\absoluteValue{f(x_n) - \sup_{I_j^n}f} < 1/n2$.
We construct $(x_n')$ similarly, but instead of the supremum, we take the infimum.
By the construction of $P_n$, we have that both $x_n \to x$ and $x_n' \to x$.
Furthermore, we have 
\begin{align*}
    \absoluteValue{\sup_{I_j^n} f - \inf_{I_j^n} f}
    =& \absoluteValue{\sup_{I_j^n} f - f(x_n) + f(x_n) - \inf_{I_j^n} f}\\
    \leq& \absoluteValue{\sup_{I_j^n} f - f(x_n)} 
        + \absoluteValue{f(x_n) - \inf_{I_j^n} f}\\
    <& 1/n,
\end{align*}
so that 
\begin{align*}
    \lim g_n(x) 
    = \lim \sum_j^{2^n} (\inf_{I_j^n} f)\chi_{I_j^n} 
    =& f^L(x)\\ 
    &= f^U(x)
    = \lim \sum_j^{2^n} (\sup_{I_j^n} f)\chi_{I_j^n}
    = \lim h_n(x),
\end{align*}
since whenever we evaluate $g_n$ or $h_n$ on $x$, the equality we want to verify is simply $\lim_{n \to \infty} \inf_{I_j^n} f = \lim_{n \to \infty} \sup_{I_j^n} f$

($\supseteq$)
We will prove this by contrapositive too.
Thus, suppose $f^L(x) = f^U(x)$ for $x \in [a,b]$, so that $\lim g_n(x) = \lim h_n(x)$.
As we argued above, this is the same as saying that for all $\epsilon > 0$, there exists $N \in \N$ such that whenever $n > N$, it holds that $\absoluteValue{\sup_{I_j^n} f - \inf_{I_j^n} f} < \epsilon$ (for the $I_j^n$ which contains $x$).
We want to prove $f$ is continuous, so that for any $x_m \to x$, we have that $f(x_m) \to f(x)$, which is the same as saying that for all $\epsilon > 0$, there exists an $M \in \N$ so that whenever $m > M$, it holds that $f(x_m) \in B_\epsilon(f(x))$.
Fix $\epsilon > 0$, and choose $N \in \N$, so that $\absoluteValue{\sup_{I_j^N} f - \inf_{I_j^N} f} < \epsilon$.
Notice that $(x_m)$ is eventually in $I_j^N$;
that is, there exists an $M \in \N$ so that whenever $m > M$, we have $x_m \in I_j^N$, and thus, $f(x_m) \in B_\epsilon(f(x))$ (as $f(x_m)$ must be within the infimum and supremum in that particular interval), completing the proof.
\end{proof} 

\begin{exercise}{9}
Suppose $(X, \cS, \mu)$ is a measure space and $E_1, E_2, \dots, E_n$ are disjoint subsets of $X$.
Suppose $a_1, \dots, a_n$ are distinct nonzero real numbers.
Prove that $f(x) = a_1 \chi_{E_1} + \dots + a_n \chi_{E_n} \in \cL^1(\mu)$ if and only if $E_k \in \cS$ and $\mu(E_k) < \infty$ for all $k \in \set{1, \dots, n}$.
Furthermore, $\norm{a_1 \chi_{E_1} + \dots + a_n \chi_{E_n}}_1 = \absoluteValue{a_1} \mu(E_1) + \dots + \absoluteValue{a_n} \mu(E_n).$
\end{exercise}
\begin{proof}
($\Rightarrow$)
If $f \in \cL^1$, then $f$ is measurable, and by exercise 2.B.13, each $E_k \in \cS$. To see $\mu(E_k) < \infty$ for all $k$, we have
\begin{align*}
    \int \absoluteValue{f} d\mu
    = \int f^+ + f^- d\mu
    = C < \infty,
\end{align*}
so that both $\int f^+ d\mu$ and $\int f^- d\mu$ must be finite.
Furthermore, since $\int f^+ d\mu = \sum_{a_k > 0} a_k \mu(E_k)$, then $\mu(E_k) < \infty$ (for all $k$ with $a_k > 0$).
Using the same technique, mutatis mutandis, we can conclude the same for all $k$ with $a_k < 0$ so that $\mu(E_k)  < \infty$, for all $k$, as required.

($\Leftarrow$)
If $f(x) \in \cL^1(\mu)$, then $f(x)$ is measurable, and by exercise 2.B.13, every $E_k \in \cS$.
We assume $\mu(E_k) < \infty$ for all $k$, implying that 
\begin{align*}
    \int f^+ d\mu
    = \int \sum_{a_i > 0} a_i \chi_{E_i} d\mu
    = \sum_{a_i > 0} a_i \mu(E_i)
    = C^+ < \infty,
\end{align*}
likewise $\int f^- d\mu = C^- < \infty$.
Putting these together, we find that $\int \absoluteValue{f} d\mu = \int f^+ + f^- d\mu = C^+ + C^- < \infty$ and $f \in \cL^1$, as required.

To see what $\norm{f}_1$ evaluates to we will use a similar technique.
We have 
\begin{align*}
    \norm{f}_1
    =& \int \absoluteValue{f} d\mu\\
    =& \int f^+ + f^- d\mu\\
    =& \int f^+ d\mu + \int f^- d\mu\\
    =& \sum_{a_i > 0} \absoluteValue{a_i} \mu(E_i) + \sum_{a_i < 0} \absoluteValue{a_i} \mu(E_i)\\
    =& \sum_i \absoluteValue{a_i} \mu(E_i)
\end{align*}
where $a_i = \absoluteValue{a_i}$, if $a_i > 0$, and $-a_i = \absoluteValue{a_i}$ if $a_i < 0$ (by the definition of $f^-$).
\end{proof} 

\begin{exercise}{10}
\begin{enumerate}
    \item Suppose $(X, \cS, \mu)$ is a measure space such that $\mu(X) < \infty$.
    Suppose $p, r$ are positive numbers with $p < r$.
    Prove that if $f: X \to [0, \infty)$ is an $\cS$-measurable function such that $\int f^r d\mu < \infty$, then $\int f^p d\mu < \infty$.
    \item Give an example to show that the previous result can be false without the hypothesis that $\mu(X) < \infty$.
\end{enumerate}
\end{exercise}
\begin{proof}
\begin{enumerate}
    \item Notice that if $f(x) < 1$, we have that $f^p(x) \leq 1$, so that $f^p(x) \leq 1 + f^r(x)$, likewise, if $f(x) \geq 1$, we have that $f^p(x) \leq f^r(x) \leq 1 + f^r(x)$.
    Thus, we have that 
    \begin{align*}
        \int f^p d\mu 
        \leq \int 1 + f^r \mu 
        = \int 1 d\mu + \int f^r d\mu
        = \mu(X) + \int f^r d\mu 
        < \infty,
    \end{align*}
    where the first inequality follows by the order preserving of Lebesgue integrals (3.22).
    \item For an example where the statement does not hold, if $\mu(X) = \infty$, consider the counting measure on $\N$, $p = 1 < 2 = r$ and the function $f(x) = 1/x$, we have $\int f^r d\mu < \infty$ but $\int f^p d\mu = \infty$.
\end{enumerate}
\end{proof}
